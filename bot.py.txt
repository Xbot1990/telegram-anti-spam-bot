# ================== –í–°–ü–û–ú–û–ì–ê–¢–ï–õ–¨–ù–´–ï ==================
def get_text(user_id, key, **kwargs):
    lang = user_language.get(user_id, 'ru')
    text = TEXTS.get(lang, {}).get(key, "")
    return text.format(**kwargs) if kwargs else text


def is_admin(user_id: int) -> bool:
    return user_id in ADMIN_IDS


def contains_bad_words(text: str) -> bool:
    if not text:
        return False
    text = text.lower()
    return any(word in text for word in bad_words)


# ================== –û–ë–†–ê–ë–û–¢–ö–ê –§–õ–£–î–ê ==================
def detect_flood(user_id: int, chat_id: int) -> bool:
    now = datetime.now()
    key = f"{chat_id}_{user_id}"
    spam_detection.setdefault(key, [])
    spam_detection[key] = [t for t in spam_detection[key] if now - t < timedelta(seconds=10)]
    spam_detection[key].append(now)
    return len(spam_detection[key]) >= 5


# ================== –ü–†–û–í–ï–†–ö–ê –°–°–´–õ–û–ö ==================
def contains_suspicious_links(text: str) -> bool:
    if not text:
        return False

    # –ò—â–µ–º –≤—Å–µ URL –≤ —Ç–µ–∫—Å—Ç–µ
    urls = re.findall(r'(?:https?://|www\.|t\.me/|bit\.ly/|tinyurl\.com/)[^\s]+', text.lower())

    for url in urls:
        # –ü—Ä–æ–≤–µ—Ä—è–µ–º, —Å–æ–¥–µ—Ä–∂–∏—Ç –ª–∏ URL —Ä–∞–∑—Ä–µ—à–µ–Ω–Ω—ã–π –¥–æ–º–µ–Ω
        is_allowed = False
        for domain in allowed_domains:
            if domain in url:
                is_allowed = True
                break

        # –ï—Å–ª–∏ –Ω–∞–π–¥–µ–Ω–∞ —Ö–æ—Ç—è –±—ã –æ–¥–Ω–∞ –∑–∞–ø—Ä–µ—â–µ–Ω–Ω–∞—è —Å—Å—ã–ª–∫–∞
        if not is_allowed:
            return True

    return False


# ================== –ù–ê–ö–ê–ó–ê–ù–ò–ï –ü–û–õ–¨–ó–û–í–ê–¢–ï–õ–Ø ==================
async def punish_user(update: Update, context: ContextTypes.DEFAULT_TYPE, reason: str):
    user_id = update.effective_user.id
    name = update.effective_user.first_name

    try:
        await update.message.delete()
    except Exception:
        pass

    user_warnings[user_id] = user_warnings.get(user_id, 0) + 1

    if user_warnings[user_id] >= 3:
        try:
            await update.effective_chat.ban_member(user_id)
            await update.effective_chat.send_message(
                get_text(user_id, 'banned', name=name, reason=reason)
            )
        except Exception as e:
            logger.error(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –±–∞–Ω–µ –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—è: {e}")
    else:
        await update.effective_chat.send_message(
            get_text(user_id, 'warned', name=name, reason=reason, count=user_warnings[user_id])
        )


# ================== –ö–û–ú–ê–ù–î–´ ==================
async def start(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(get_text(update.effective_user.id, 'start'))


async def help_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(get_text(update.effective_user.id, 'help'))


async def language_command(update: Update, context: ContextTypes.DEFAULT_TYPE):
    await update.message.reply_text(get_text(update.effective_user.id, 'choose_lang'))


async def set_language(update: Update, context: ContextTypes.DEFAULT_TYPE):
    text = update.message.text.lower()
    user_id = update.effective_user.id

    if "english" in text or "–∞–Ω–≥–ª–∏–π—Å–∫–∏–π" in text:
        user_language[user_id] = 'en'
        await update.message.reply_text(get_text(user_id, 'lang_changed'))
    elif "—Ä—É—Å—Å–∫–∏–π" in text or "russian" in text:
        user_language[user_id] = 'ru'
        await update.message.reply_text(get_text(user_id, 'lang_changed'))


# ================== –ü–†–ò–í–ï–¢–°–¢–í–ò–ï ==================
async def welcome_new_member(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if not update.message or not update.message.new_chat_members:
        return

    for new_member in update.message.new_chat_members:
        if new_member.is_bot and new_member.id != context.bot.id:
            try:
                await update.message.reply_text(
                    get_text(update.effective_user.id, 'no_bots')
                )
                await update.effective_chat.ban_member(new_member.id)
            except Exception as e:
                logger.error(f"–û—à–∏–±–∫–∞ –ø—Ä–∏ –±–∞–Ω–µ –±–æ—Ç–∞: {e}")
            return

        await update.message.reply_text(
            get_text(update.effective_user.id, 'welcome', name=new_member.first_name)
        )


# ================== –ê–ù–ê–õ–ò–ó –°–û–û–ë–©–ï–ù–ò–ô ==================
async def analyze_message(update: Update, context: ContextTypes.DEFAULT_TYPE):
    user_id = update.effective_user.id
    chat_id = update.effective_chat.id
    text = update.message.text or update.message.caption or ""

    if is_admin(user_id):
        return

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Ñ–ª—É–¥
    if detect_flood(user_id, chat_id):
        await punish_user(update, context, "–§–ª—É–¥")
        return

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –º–∞—Ç
    if contains_bad_words(text):
        await punish_user(update, context, "–ú–∞—Ç")
        return

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ —Å–ø–∞–º-–∫–ª—é—á–µ–≤—ã–µ —Å–ª–æ–≤–∞
    if any(word in text.lower() for word in spam_keywords):
        await punish_user(update, context, "–°–ø–∞–º")
        return

    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –Ω–∞ –ø–æ–¥–æ–∑—Ä–∏—Ç–µ–ª—å–Ω—ã–µ —Å—Å—ã–ª–∫–∏
    if contains_suspicious_links(text):
        await punish_user(update, context, "–°—Å—ã–ª–∫–∞")
        return


# ================== –ê–ù–ê–õ–ò–ó –ú–ï–î–ò–ê ==================
async def analyze_media(update: Update, context: ContextTypes.DEFAULT_TYPE):
    if is_admin(update.effective_user.id):
        return

    try:
        await update.message.delete()
    except Exception:
        pass

    await update.effective_chat.send_message(
        get_text(update.effective_user.id, 'no_media', name=update.effective_user.first_name)
    )

def main():
    app = Application.builder().token(TOKEN).build()

    # –û–±—Ä–∞–±–æ—Ç—á–∏–∫–∏ –∫–æ–º–∞–Ω–¥
    app.add_handler(CommandHandler("start", start))

    app.add_handler(CommandHandler("language", language_command))

        filters.TEXT & filters.Regex(r'(?i)(english|—Ä—É—Å—Å–∫–∏–π|russian|–∞–Ω–≥–ª–∏–π—Å–∫–∏–π)'),
        set_language

    app.add_handler(MessageHandler(filters.StatusUpdate.NEW_CHAT_MEMBERS, welcome_new_member))
    app.add_handler(MessageHandler(filters.TEXT & ~filters.COMMAND, analyze_message))
    app.add_handler(MessageHandler(
        filters.PHOTO | filters.VIDEO | filters.Document.ALL | filters.AUDIO | filters.VOICE,
        analyze_media
    ))

    print("ü§ñ Anti-Spam Bot –∑–∞–ø—É—â–µ–Ω...")
